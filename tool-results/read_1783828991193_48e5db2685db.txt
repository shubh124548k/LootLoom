    72→  return name
    73→    .split(" ")
    74→    .map((p) => p[0])

    76→    .slice(0, 2)
    77→    .join("")
    78→    .toUpperCase();
    79→}

   326→function AchievementCard({ item, index }: { item: AchievementItem; index: number }) {
   327→  const meta = RARITY_META[item.rarity];
   328→  const pct = Math.min(100, (item.progress / item.total) * 100);

   332→    <motion.div variants={cardReveal} custom={index}>
   333→      <GlassCard
   334→        level={2}
   335→        hover
   336→        sheen
   337→        className={cn(
   338→          "p-5 h-full flex flex-col items-center text-center ring-1",
   339→          item.unlocked ? meta.glow : "ring-border opacity-80"
   340→        )}
   341→      >
   342→        <div className="relative mb-4">
   343→          <ProgressRing
   344→            value={pct}
   345→            size={88}
   346→            strokeWidth={6}
   347→            gradient={meta.ring}
   348→            showLabel={false}
   349→          />
   350→          <div
   351→            className={cn(
   352→              "absolute inset-0 m-auto size-14 rounded-2xl flex items-center justify-center ring-1",
   353→              item.unlocked
   354→                ? "glass-3 " + meta.glow
   355→                : "glass-1 ring-border"
   356→            )}
   357→          >
   358→            {item.unlocked ? (
   359→              <LucideIcon className={cn("text-foreground")} size={22} />
   360→            ) : (
   361→              <Icons.Lock className="text-muted-foreground" size={20} />
   362→            )}
   363→          </div>
   364→        </div>
   365→        <h3 className="font-semibold text-sm text-foreground mb-1">{item.name}</h3>
   366→        <p className="text-xs text-muted-foreground mb-3 leading-relaxed">{item.description}</p>
   367→        <div className="flex items-center gap-2 mt-auto">
   368→          <StatusBadge
   369→            variant={
   370→              item.rarity === "common"
   371→                ? "cyan"
   372→                : item.rarity === "rare"
   373→                ? "electric"
   374→                : item.rarity === "epic"
   375→                ? "purple"
   376→                : "gold"
   377→            }
   378→            dot
   379→          >
   380→            {meta.label}

   389→            </span>
   390→          )}
   391→        </div>
   392→      </GlassCard>
   393→    </motion.div>
   394→  );
