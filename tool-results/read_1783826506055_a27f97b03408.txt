   255→function getInitials(name: string): string {
   256→  return name
   257→    .split(/\s+/)
   258→    .filter(Boolean)
   259→    .slice(0, 2)
   260→    .map((s) => s[0]?.toUpperCase() ?? "")
   261→    .join("");
   262→}
   263→
   264→function useCopy() {
   265→  const [copied, setCopied] = useState(false);
   266→  const copy = async (text: string) => {
   267→    try {
   268→      if (typeof navigator !== "undefined" && navigator?.clipboard?.writeText) {
   269→        await navigator.clipboard.writeText(text);
   270→      }
   271→    } catch {
   272→      /* no-op */
   273→    }
   274→    setCopied(true);
   275→    window.setTimeout(() => setCopied(false), 1800);
   276→  };
   277→  return { copied, copy };
   278→}
